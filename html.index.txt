    <script>
        // DOM Elements
        const functionInput = document.getElementById('functionInput');
        const limitAInput = document.getElementById('limitA');
        const limitBInput = document.getElementById('limitB');
        const intervalsInput = document.getElementById('intervals');
        const resultValue = document.getElementById('resultValue');
        const errorMsg = document.getElementById('errorMsg');
        const ctx = document.getElementById('graphCanvas').getContext('2d');

        let myChart = null;

        // Convert standard math string configurations into executable JS Math syntax
        function parseExpression(str) {
            let processed = str.toLowerCase();
            // Replace basic powers e.g., x^2 -> Math.pow(x,2). Handles simple x^digit variables.
            processed = processed.replace(/([x\d\)]+)\s*\^\s*([\d\.]+)/g, 'Math.pow($1, $2)');
            // Replace recognized math functions
            const functions = ['sin', 'cos', 'tan', 'log', 'exp', 'sqrt', 'abs', 'PI', 'E'];
            functions.forEach(f => {
                const regex = new RegExp(`\\b${f}\\b`, 'g');
                processed = processed.replace(regex, `Math.${f}`);
            });
            return processed;
        }

        // Evaluate the function string safely at a specific point 'x'
        function evaluateFunction(parsedStr, x) {
            try {
                // Creates a sandboxed evaluator bound to x
                const evaluator = new Function('x', `return ${parsedStr};`);
                const res = evaluator(x);
                return isNaN(res) || !isFinite(res) ? null : res;
            } catch (e) {
                return null;
            }
        }

        // Perform numerical definite integration using the Trapezoidal Rule
        function calculateIntegral(parsedStr, a, b, n) {
            const h = (b - a) / n;
            let sum = 0.5 * (evaluateFunction(parsedStr, a) + evaluateFunction(parsedStr, b));
            
            for (let i = 1; i < n; i++) {
                let x = a + i * h;
                sum += evaluateFunction(parsedStr, x);
            }
            
            return sum * h;
        }

        // Render line chart with the integration window shaded
        function updateCalculator() {
            errorMsg.style.display = 'none';
            const rawExpr = functionInput.value;
            const parsedExpr = parseExpression(rawExpr);
            
            const a = parseFloat(limitAInput.value);
            const b = parseFloat(limitBInput.value);
            const n = parseInt(intervalsInput.value);

            if (isNaN(a) || isNaN(b) || isNaN(n)) return;

            // Validate expression by testing a value
            if (evaluateFunction(parsedExpr, a) === null && evaluateFunction(parsedExpr, (a+b)/2) === null) {
                errorMsg.style.display = 'block';
                return;
            }

            // Calculate integral
            const area = calculateIntegral(parsedExpr, a, b, n);
            resultValue.innerText = area.toFixed(5);

            // Generate analytical layout scale bounds for graphing
            const padding = Math.max(Math.abs(b - a) * 0.5, 2);
            const minX = Math.min(a, b) - padding;
            const maxX = Math.max(a, b) + padding;
            
            const graphPoints = 150;
            const step = (maxX - minX) / graphPoints;

            const lineData = [];
            const integralData = [];

            for (let i = 0; i <= graphPoints; i++) {
                const x = minX + (i * step);
                const y = evaluateFunction(parsedExpr, x);
                
                if (y !== null) {
                    lineData.push({ x: x, y: y });
                    
                    // Populate area filling structure if coordinate falls inside boundaries [a, b]
                    if (x >= Math.min(a, b) && x <= Math.max(a, b)) {
                        integralData.push({ x: x, y: y });
                    }
                }
            }

            // Chart.js Configuration
            const data = {
                datasets: [
                    {
                        label: `f(x) = ${rawExpr}`,
                        data: lineData,
                        borderColor: '#1e293b',
                        borderWidth: 2,
                        pointRadius: 0,
                        fill: false,
                        tension: 0.1
                    },
                    {
                        label: 'Integrated Area',
                        data: integralData,
                        backgroundColor: 'rgba(37, 99, 235, 0.25)',
                        borderColor: 'rgba(37, 99, 235, 0.6)',
                        borderWidth: 1,
                        pointRadius: 0,
                        fill: 'origin', // shade to x-axis
                    }
                ]
            };

            if (myChart) {
                myChart.data = data;
                myChart.options.scales.x.min = minX;
                myChart.options.scales.x.max = maxX;
                myChart.update('none'); // Update without heavy animations for a snappier feel
            } else {
                myChart = new Chart(ctx, {
                    type: 'line',
                    data: data,
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        scales: {
                            x: {
                                type: 'linear',
                                position: 'bottom',
                                min: minX,
                                max: maxX,
                                grid: { color: '#e2e8f0' }
                            },
                            y: {
                                type: 'linear',
                                grid: { color: '#e2e8f0' }
                            }
                        },
                        plugins: {
                            legend: { display: false }
                        }
                    }
                });
            }
        }

        // Add real-time user action event listeners
        [functionInput, limitAInput, limitBInput, intervalsInput].forEach(element => {
            element.addEventListener('input', updateCalculator);
        });

        // Initialize display configuration on load
        updateCalculator();
    </script>
